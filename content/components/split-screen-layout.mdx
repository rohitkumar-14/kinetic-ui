## Overview
A foundational layout component commonly used on Awwwards sites for product features, dual narratives, or sticky text paired with scrolling images. 

## Usage

```tsx
import { SplitScreenLayout } from "@/components/creative/split-screen-layout"
```

```tsx
<SplitScreenLayout
  splitRatio="40/60"
  stickySide="left"
  leftContent={
    <div className="p-12 flex flex-col justify-center h-full">
      <h2>Sticky Heading</h2>
      <p>This side stays pinned to the screen.</p>
    </div>
  }
  rightContent={
    <div className="p-12 space-y-24">
      <img src="/img1.jpg" alt="Image 1" />
      <img src="/img2.jpg" alt="Image 2" />
      <img src="/img3.jpg" alt="Image 3" />
    </div>
  }
/>
```
